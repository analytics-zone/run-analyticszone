ટેકનિકલ સંદર્ભ અને સૂત્રો (Formulas)

સંપૂર્ણ ગાણિતિક અમલીકરણ (Mathematical Implementation)

અમલીકરણ માર્ગદર્શિકા

આ પેજ તમામ રન એનાલિટિક્સ મેટ્રિક્સ માટે કોપી-પેસ્ટ કરી શકાય તેવા સૂત્રો અને સ્ટેપ-બાય-સ્ટેપ ગણતરીની પદ્ધતિઓ પ્રદાન કરે છે. કસ્ટમ અમલીકરણ, ચકાસણી અથવા ઊંડી સમજણ માટે આનો ઉપયોગ કરો.

⚠️ અમલીકરણ નોંધો

  • ગણતરી માટે તમામ સમયને સેકન્ડમાં રૂપાંતરિત કરવા જોઈએ
  • rTSS ની ગણતરીમાં ઇન્ટેન્સિટી ફેક્ટરના વર્ગ (IF²) નો ઉપયોગ થાય છે
  • હંમેશા ઇનપુટ્સની વ્યાજબી રેન્જ માટે તપાસ કરો
  • એજ કેસ (શૂન્ય વડે ભાગાકાર, નેગેટિવ મૂલ્યો) ને હેન્ડલ કરો

મુખ્ય પર્ફોર્મન્સ મેટ્રિક્સ

ક્રિટિકલ રન સ્પીડ (CRS)

સૂત્ર:

CRS (m/s) = (D₂ - D₁) / (T₂ - T₁)
CRS પેસ (min/km) = 16.667 / CRS (m/s)

🧪 ઇન્ટરેક્ટિવ કેલ્ક્યુલેટર - સૂત્રનું પરીક્ષણ કરો

CRS પેસ (min/km):
4:17
ગણતરીના પગલાં:
CRS (m/s) = (3600 - 1200) / (900 - 270) = 3.81 m/s
પેસ (min/km) = 1000 / 3.81 = 262 સેકન્ડ = 4:22

JavaScript અમલીકરણ:

function calculateCRS(distance1, time1, distance2, time2) {
  // સમયને સેકન્ડમાં રૂપાંતરિત કરો
  const t1 = typeof time1 === 'string' ? timeToSeconds(time1) : time1;
  const t2 = typeof time2 === 'string' ? timeToSeconds(time2) : time2;

  // CRS વેલોસિટી m/s માં ગણો
  const crs_ms = (distance2 - distance1) / (t2 - t1);

  // પેસ પ્રતિ કિલોમીટર સેકન્ડમાં ગણો
  const pace_per_km = 1000 / crs_ms;

  // mm:ss ફોર્મેટમાં રૂપાંતરિત કરો
  const minutes = Math.floor(pace_per_km / 60);
  const seconds = Math.round(pace_per_km % 60);

  return {
    velocity_ms: crs_ms,
    pace_seconds: pace_per_km,
    pace_formatted: `${minutes}:${seconds.toString().padStart(2, '0')}`
  };
}

રનિંગ સ્ટ્રેસ સ્કોર (rTSS)

સંપૂર્ણ સૂત્ર:

rTSS = (IF²) × અવધિ (કલાક) × 100
IF = NSS / થ્રેશોલ્ડ સ્પીડ
NSS = કુલ અંતર / કુલ સમય (m/min)

🧪 ઇન્ટરેક્ટિવ કેલ્ક્યુલેટર - સૂત્રનું પરીક્ષણ કરો

ગણેલ rTSS:
31
ગણતરીના પગલાં:
NSS = 5000m / 25min = 200 m/min
FTP = 1000 / (255/60) = 235.3 m/min
IF = 200 / 235.3 = 0.850
rTSS = 0.850² × (25/60) × 100 = 31

JavaScript અમલીકરણ:

function calculateRTSS(distance, timeMinutes, ftpMetersPerMin) {
  // નોર્મલાઈઝ્ડ રન સ્પીડ ગણો
  const nss = distance / timeMinutes;

  // ઇન્ટેન્સિટી ફેક્ટર ગણો
  const intensityFactor = nss / ftpMetersPerMin;

  // કલાકો ગણો
  const hours = timeMinutes / 60;

  // ક્વોડ્રેટિક ઇન્ટેન્સિટી ફેક્ટરનો ઉપયોગ કરીને rTSS ગણો
  const rtss = Math.pow(intensityFactor, 2) * hours * 100;

  return Math.round(rtss);
}

// વપરાશનું ઉદાહરણ:
const rtss = calculateRTSS(3000, 55, 64.5);
// રીટર્ન: 65

// હેલ્પર: CRS પેસને રનિંગ સ્પીડ (m/min) માં રૂપાંતરિત કરો
function crsPaceToSpeed(crsPacePerKmSeconds) {
  // m/min માં સ્પીડ = 1000m / (મિનિટમાં પેસ)
  return 1000 / (crsPacePerKmSeconds / 60);
}

// ઉદાહરણ: 4:15 નું CRS (255 સેકન્ડ)
const speed = crsPaceToSpeed(255); // રીટર્ન: 235.3 m/min

બાયોમિકેનિકલ કાર્યક્ષમતા: વર્ટિકલ રેશિયો

સૂત્ર:

વર્ટિકલ રેશિયો (%) = (વર્ટિકલ ઓસિલેશન ÷ સ્ટ્રાઈડ લેન્થ) × 100

🧪 ઇન્ટરેક્ટિવ કેલ્ક્યુલેટર - સૂત્રનું પરીક્ષણ કરો

વર્ટિકલ રેશિયો:
7.4%
ગણતરી:
વર્ટિકલ રેશિયો = (8.5 / 115) × 100 = 7.4%

JavaScript અમલીકરણ:

function calculateVerticalRatio(oscillationCm, strideLengthCm) {
  return (oscillationCm / strideLengthCm) * 100;
}

function calculateEfficiencyFactor(paceMetersPerMin, avgHeartRate) {
  return paceMetersPerMin / avgHeartRate;
}

ચાલી રહી છે સ્ટ્રાઈડ મિકેનિક્સ

સ્ટ્રાઈડ રેટ (SR)

સૂત્ર:

SR = 60 / સાયકલ ટાઈમ (સેકન્ડ)
SR = (સ્ટ્રાઈડ્સની સંખ્યા / સેકન્ડમાં સમય) × 60

🧪 ઇન્ટરેક્ટિવ કેલ્ક્યુલેટર - સૂત્રનું પરીક્ષણ કરો

કેડન્સ (SPM):
180
ગણતરી:
કેડન્સ = (180 / 60) × 60 = 180 SPM

JavaScript અમલીકરણ:

function calculateCadence(stepCount, timeSeconds) {
  return (stepCount / timeSeconds) * 60;
}

// ઉદાહરણ:
const spm = calculateCadence(180, 60);
// રીટર્ન: 180 SPM

સ્ટ્રાઈડ લેન્થ (Stride Length)

સૂત્ર:

સ્ટ્રાઈડ લેન્થ = અંતર / (સ્ટેપ કાઉન્ટ / 2)
સ્ટ્રાઈડ લેન્થ = વેલોસિટી / (કેડન્સ / 120)

JavaScript અમલીકરણ:

function calculateStrideLength(distanceMeters, stepCount) {
  // સ્ટ્રાઈડ લેન્થ એ અંતર / સ્ટ્રાઈડ જોડીઓની સંખ્યા (ડાબો + જમણો) છે
  return distanceMeters / (stepCount / 2);
}

// ઉદાહરણ (1000m, 800 સ્ટેપ્સ):
const strideLength = calculateStrideLength(1000, 800);
// રીટર્ન: 2.50 મીટર પ્રતિ સ્ટ્રાઈડ

SR અને DPS પરથી વેલોસિટી

સૂત્ર:

વેલોસિટી (m/s) = (SR / 60) × DPS

JavaScript અમલીકરણ:

function calculateVelocity(strideRate, dps) {
  return (strideRate / 60) * dps;
}

// ઉદાહરણ:
const velocity = calculateVelocity(70, 1.6);
// રીટર્ન: 1.87 m/s

સ્ટ્રાઈડ ઇન્ડેક્સ (SI)

સૂત્ર:

SI = વેલોસિટી (m/s) × DPS (m/stride)

JavaScript અમલીકરણ:

function calculateStrideIndex(velocity, dps) {
  return velocity * dps;
}

// ઉદાહરણ:
const si = calculateStrideIndex(1.5, 1.7);
// રીટર્ન: 2.55

પર્ફોર્મન્સ મેનેજમેન્ટ ચાર્ટ (PMC) દોડવા માટે

CTL, ATL, TSB ગણતરીઓ

સૂત્રો:

CTL આજે = CTL ગઈકાલે + (TSS આજે - CTL ગઈકાલે) × (1/42)
ATL આજે = ATL ગઈકાલે + (TSS આજે - ATL ગઈકાલે) × (1/7)
TSB = CTL ગઈકાલે - ATL ગઈકાલે

JavaScript અમલીકરણ:

function updateCTL(previousCTL, todayTSS) {
  return previousCTL + (todayTSS - previousCTL) * (1/42);
}

function updateATL(previousATL, todayTSS) {
  return previousATL + (todayTSS - previousATL) * (1/7);
}

function calculateTSB(yesterdayCTL, yesterdayATL) {
  return yesterdayCTL - yesterdayATL;
}

// વર્કઆઉટ્સની શ્રેણી માટે PMC ગણો
function calculatePMC(workouts) {
  let ctl = 0, atl = 0;
  const results = [];

  workouts.forEach(workout => {
    ctl = updateCTL(ctl, workout.tss);
    atl = updateATL(atl, workout.tss);
    const tsb = calculateTSB(ctl, atl);

    results.push({
      date: workout.date,
      tss: workout.tss,
      ctl: Math.round(ctl * 10) / 10,
      atl: Math.round(atl * 10) / 10,
      tsb: Math.round(tsb * 10) / 10
    });
  });

  return results;
}

// વપરાશનું ઉદાહરણ:
const workouts = [
  { date: '2025-01-01', tss: 50 },
  { date: '2025-01-02', tss: 60 },
  { date: '2025-01-03', tss: 45 },
  // ... વધુ વર્કઆઉટ્સ
];

const pmc = calculatePMC(workouts);
// દરેક દિવસ માટે CTL, ATL, TSB સાથેનો એરે રીટર્ન કરે છે
અમારી સંપૂર્ણ તાલીમ લોડ માર્ગદર્શિકા વાંચો →

એડવાન્સ્ડ ગણતરીઓ

બહુવિધ અંતરો પરથી CRS (રીગ્રેશન પદ્ધતિ)

JavaScript અમલીકરણ:

function calculateCRSRegression(distances, times) {
  // લીનિયર રીગ્રેશન: અંતર = a + b*સમય
  const n = distances.length;
  const sumX = times.reduce((a, b) => a + b, 0);
  const sumY = distances.reduce((a, b) => a + b, 0);
  const sumXY = times.reduce((sum, x, i) => sum + x * distances[i], 0);
  const sumXX = times.reduce((sum, x) => sum + x * x, 0);

  const slope = (n * sumXY - sumX * sumY) / (n * sumXX - sumX * sumX);
  const intercept = (sumY - slope * sumX) / n;

  return {
    crs: slope, // ક્રિટિકલ રનિંગ વેલોસિટી (m/s)
    anaerobic_capacity: intercept // એનરોબિક અંતર ક્ષમતા (m)
  };
}

// બહુવિધ પરીક્ષણ અંતરો સાથેનું ઉદાહરણ:
const distances = [100, 200, 400, 800];
const times = [65, 150, 340, 720]; // સેકન્ડમાં
const result = calculateCRSRegression(distances, times);
// રીટર્ન: { crs: 1.18, anaerobic_capacity: 15.3 }

પેસ પરથી ઇન્ટેન્સિટી ફેક્ટર

JavaScript અમલીકરણ:

function calculateIntensityFactor(actualPaceMinKm, thresholdPaceMinKm) {
  // જો જરૂરી હોય તો પેસ સ્ટ્રિંગ્સ "mm:ss" ને સેકન્ડમાં રૂપાંતરિત કરો
  const actualSecs = typeof actualPaceMinKm === 'string' ? timeToSeconds(actualPaceMinKm) : actualPaceMinKm;
  const thresholdSecs = typeof thresholdPaceMinKm === 'string' ? timeToSeconds(thresholdPaceMinKm) : thresholdPaceMinKm;

  // IF એ થ્રેશોલ્ડ પેસ / વાસ્તવિક પેસ છે (ઝડપી પેસ = ઓછી સેકન્ડ વેલ્યુ)
  return thresholdSecs / actualSecs;
}

// ઉદાહરણ:
const if_value = calculateIntensityFactor("4:45", "4:15");
// રીટર્ન: 0.895 (થ્રેશોલ્ડના 89.5% પર દોડી રહ્યા છો)

પેસ સાતત્ય (Consistency) વિશ્લેષણ

JavaScript અમલીકરણ:

function analyzePaceConsistency(segments) {
  const paces = segments.map(kilometer => kilometer.distance / kilometer.time);
  const avgPace = paces.reduce((a, b) => a + b) / paces.length;

  const variance = paces.reduce((sum, pace) =>
    sum + Math.pow(pace - avgPace, 2), 0) / paces.length;
  const stdDev = Math.sqrt(variance);
  const coefficientOfVariation = (stdDev / avgPace) * 100;

  return {
    avgPace,
    stdDev,
    coefficientOfVariation,
    consistency: coefficientOfVariation < 5 ? "Excellent" :
                 coefficientOfVariation < 10 ? "Good" :
                 coefficientOfVariation < 15 ? "Moderate" : "Variable"
  };
}

// ઉદાહરણ:
const segments = [
  { distance: 100, time: 70 },
  { distance: 100, time: 72 },
  { distance: 100, time: 71 },
  // ...
];
const analysis = analyzePaceConsistency(segments);
// રીટર્ન: { avgPace: 1.41, stdDev: 0.02, coefficientOfVariation: 1.4, consistency: "Excellent" }

સ્ટ્રાઈડ એફિશિયન્સી પરથી થાકની ઓળખ

JavaScript અમલીકરણ:

function detectFatigue(segments) {
  // પ્રથમ અને છેલ્લા સેગમેન્ટના વર્ટિકલ રેશિયોની સરખામણી કરો
  const firstSegment = segments[0];
  const lastSegment = segments[segments.length - 1];

  const ratioIncrease = lastSegment.verticalRatio - firstSegment.verticalRatio;

  return {
    startRatio: firstSegment.verticalRatio,
    endRatio: lastSegment.verticalRatio,
    increase: Math.round(ratioIncrease * 10) / 10,
    fatigueLevel: ratioIncrease < 0.2 ? "Minimal" :
                  ratioIncrease < 0.5 ? "Moderate" :
                  ratioIncrease < 1.0 ? "Significant" : "Severe"
  };
}

// ઉદાહરણ:
const segments = [
  { verticalRatio: 7.2 }, { verticalRatio: 7.3 }, { verticalRatio: 8.1 }
];
const fatigue = detectFatigue(segments);
// રીટર્ન: { startRatio: 7.2, endRatio: 8.1, increase: 0.9, fatigueLevel: "Significant" }

ડેટા વેરિફિકેશન (Data Validation)

વર્કઆઉટ ડેટા ક્વોલિટી ચેક્સ

JavaScript અમલીકરણ:

function validateWorkoutData(workout) {
  const issues = [];

  // વ્યાજબી પેસ રેન્જ માટે તપાસ (3:00-8:00 પ્રતિ કિમી)
  const avgPaceSecs = workout.totalTime / (workout.totalDistance / 1000);
  if (avgPaceSecs < 180 || avgPaceSecs > 480) {
    issues.push(`અસામાન્ય સરેરાશ પેસ: ${Math.round(avgPaceSecs)}સે પ્રતિ કિમી`);
  }

  // વ્યાજબી વર્ટિકલ રેશિયો માટે તપાસ (4% - 15%)
  if (workout.avgVerticalRatio < 4 || workout.avgVerticalRatio > 15) {
    issues.push(`અસામાન્ય વર્ટિકલ રેશિયો: ${workout.avgVerticalRatio}%`);
  }

  // વ્યાજબી કેડન્સ માટે તપાસ (120-220 SPM)
  const avgCadence = calculateCadence(workout.totalSteps, workout.totalTime);
  if (avgCadence < 120 || avgCadence > 220) {
    issues.push(`અસામાન્ય કેડન્સ: ${Math.round(avgCadence)} SPM`);
  }

  // ખૂટતા સેગમેન્ટ્સ માટે તપાસ (સમયમાં ગેપ)
  if (workout.segments && workout.segments.length > 1) {
    for (let i = 1; i < workout.segments.length; i++) {
      const gap = workout.segments[i].startTime -
                  (workout.segments[i-1].startTime + workout.segments[i-1].duration);
      if (gap > 300) { // 5 મિનિટનો ગેપ
        issues.push(`સેગમેન્ટ્સ ${i} અને ${i+1} વચ્ચે મોટો ગેપ જોવા મળ્યો`);
      }
    }
  }

  return {
    isValid: issues.length === 0,
    issues
  };
}

const workout = {
  totalDistance: 2000,
  totalTime: 1800, // 30 મિનિટ
  totalStrides: 800,
  segments: [/* કિલોમીટર ડેટા */]
};
const validation = validateWorkoutData(workout);

સમય રૂપાંતરણ (Time Conversion) યુટિલિટીઝ

JavaScript અમલીકરણ:

// mm:ss ને સેકન્ડમાં રૂપાંતરિત કરો
function timeToSeconds(timeString) {
  const parts = timeString.split(':');
  return parseInt(parts[0]) * 60 + parseInt(parts[1]);
}

// સેકન્ડને mm:ss માં રૂપાંતરિત કરો
function secondsToTime(seconds) {
  const minutes = Math.floor(seconds / 60);
  const secs = Math.round(seconds % 60);
  return `${minutes}:${secs.toString().padStart(2, '0')}`;
}
